public class UseFactSum {
                public static void main(String[] args) {
                    Scanner kb=new Scanner(System.in);
                    int n;
                    System.out.println("Enter a number:");
                    n=kb.nextInt();
                    int sum=0;
                    FactSum obj=new FactSum();
                    for(int i=1;i<=n;i++)
                    {
                        sum=sum+ obj.fact(i);
                    }
                    System.out.println("Sum of Factorial="+sum);
                }
